\section*{Problem Statement \& Motivation}
\textbf{Problem.} Language models can produce different generations for prompts that differ only by a demographic mention
(e.g., gender, religion, nationality). These differences can manifest as \textbf{social bias} in open-ended generation.
We adopt the ``\textbf{regard}'' framing: how positively/negatively a generation \textit{portrays} the demographic.

\textbf{Open question we study.} \textit{Does the decoding algorithm (greedy / temperature / top-$k$ / top-$p$) amplify or mitigate bias?}
Even with a fixed model checkpoint, decoding changes diversity and repetition; it may also shift the \textbf{distribution of regard}.

\textbf{Why it matters.}
\begin{itemize}
  \item Practical: decoding is a deployment-time knob (no retraining) that might reduce harmful outputs.
  \item Scientific: separates ``model bias'' from ``sampling-induced bias''.
  \item Feasible: end-to-end on Kaggle notebooks with controlled experiments and clear metrics.
\end{itemize}

\section*{Data}
\textbf{Data source(s).}
\begin{itemize}
  \item \textbf{Regard-labeled dataset \& tools:} \texttt{ewsheng/nlg-bias} (\texttt{data/regard/*.tsv}; label in col1, text in col2).
  \item \textbf{Prompt templates:} paper-style templates (e.g., occupation/descriptors) with demographic slot; we will use a fixed prompt bank for reproducibility.
  \item \textbf{Generated samples:} GPT-2 completions per (prompt type $\times$ demographic $\times$ decoding).
\end{itemize}

\textbf{License / usage rights.}
\begin{itemize}
  \item \textbf{GPT-2} and \textbf{BERT}: use released checkpoints under their published licenses/terms (cite in References).
  \item \textbf{nlg-bias repo:} if no explicit license is present, treat as research artifact: use with attribution; do not redistribute beyond class submission.
\end{itemize}

\textbf{Preprocessing steps.}
\begin{itemize}
  \item \textbf{Mask demographics for scoring:} replace demographic mention with \texttt{XYZ} before running regard classifier (standard in nlg-bias workflow).
  \item Tokenize with BERT tokenizer; max length 128 (truncate/pad).
  \item Generation: fixed max new tokens; filter empty outputs; log prompts, seeds, and decoding config.
\end{itemize}

\section*{Baseline Plan}
\textbf{Baseline (first implementation).}
\begin{itemize}
  \item \textbf{Generator:} pretrained \textbf{GPT-2 small} (no fine-tuning).
  \item \textbf{Decoding:} \textbf{greedy} decoding only.
  \item \textbf{Bias scoring:} use \textbf{released regard classifier} from nlg-bias to label generations and compute group-level bias.
\end{itemize}

\textbf{Core extension (main contribution).}
\begin{itemize}
  \item \textbf{Decoding sweep:} temperature/top-$k$/top-$p$ (+ optional anti-repetition) and measure how \textbf{regard gaps} shift.
  \item \textbf{Classifier replication (optional but strong):} fine-tune a single BERT regard classifier on the released labeled set and compare to released model (accuracy + agreement).
\end{itemize}

\textbf{Fairness constraints.}
\begin{itemize}
  \item Same checkpoint, same prompt bank, same \#samples, same max length across decoding settings.
  \item Multiple seeds per config; report variance.
\end{itemize}

\section*{Evaluation Plan}
\textbf{Primary bias metrics.}
\begin{itemize}
  \item \textbf{Regard distribution per group:} $P(\text{neg}),P(\text{neu}),P(\text{pos}),P(\text{other})$.
  \item \textbf{Negative-regard gap:} $\Delta_{\text{neg}}$ between groups within the same prompt type.
  \item \textbf{Uncertainty:} bootstrap confidence intervals over prompts/samples.
\end{itemize}

\textbf{Generation quality controls (interpretation).}
\begin{itemize}
  \item Diversity: distinct-1/2 (unique $n$-grams / total).
  \item Degeneration: repeated $n$-gram rate; longest repetition span.
  \item (Optional) toxicity proxy score (aggregate only).
\end{itemize}

\textbf{Protocol.}
\begin{itemize}
  \item Fixed prompt bank (30--80 prompts) $\times$ selected demographics (keep scope manageable).
  \item Samples per prompt/demographic: $N=50$ (scale to Kaggle budget); 3 seeds per decoding config.
  \item Score masked text; run small ablation: masked vs unmasked scoring to test sensitivity.
\end{itemize}

\section*{Risks}
\textbf{Compute limitations.}
\begin{itemize}
  \item Kaggle runtime/quota limits may cap $N$ or grid size.
  \item Mitigation: cache generations; prioritize high-signal configs (greedy, one temp, one top-$k$, one top-$p$).
\end{itemize}

\textbf{Data quality concerns.}
\begin{itemize}
  \item Regard is subjective; classifier error can skew gaps.
  \item Mitigation: CIs + manual spot-check (20--40 examples) + classifier agreement if we fine-tune.
\end{itemize}

\textbf{Potential failure modes.}
\begin{itemize}
  \item Bias shifts due to verbosity/length changes under decoding.
  \item Mitigation: fix max length; normalize by length; compare per-prompt rather than only global aggregates.
\end{itemize}

\section*{Ethics \& Safety Considerations}

\textbf{Privacy / sensitivity.}

\end{itemize}

\section*{What is happening inside?}
\textbf{Applicability.} We do \textbf{not} fine-tune a very large generator; we use a small pretrained LM and (optionally) fine-tune a classifier.

\textbf{Model architecture.}
\begin{itemize}
  \item \textbf{Generator:} GPT-2 (decoder-only Transformer, causal self-attention).
  \item \textbf{Regard classifier:} BERT-base (encoder-only Transformer) + linear head (4-way).
\end{itemize}

\textbf{Training objective \& details.}
\begin{itemize}
  \item Generator: no training; controlled decoding only.
  \item Classifier: cross-entropy; AdamW; early stopping on dev metric; report batch size/LR/epochs and best checkpoint.
\end{itemize}
